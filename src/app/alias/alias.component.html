# cherry pick
git config --global alias.cp cherry-pick

# status
git config --global alias.s status

# fetch
git config --global alias.f fetch
git config --global alias.fa "fetch --all"
git config --global alias.ft "fetch --tags"

# log
git config --global alias.last 'log -1'
git config --global alias.l 'log'
git config --global alias.lg 'log --graph'
git config --global alias.lcompact "log --pretty='%h - %s AUTHOR %an COMMITTER %cn' --since='2008-10-01' --before='2018-11-01'"
git config --global alias.ldetail 'log --pretty=format:"[%h] %ae, %ar: %s" --stat'
git config --global alias.lgdetail "log --graph --abbrev-commit --decorate --format=format:'%C(bold blue)%h%C(reset) - %C(bold green)(%ar)%C(reset) %C(white)%s%C(reset) %C(dim white)- %an%C(reset)%C(bold yellow)%d%C(reset)' --all"
